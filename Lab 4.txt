import pandas as pd
import numpy as np
data=pd.read_csv("C:\\Use

wnloads\\enjoysport.csv")
concept=np.array(data)[:,:-1]
target=np.array(data)[:,-1]
h=concept[0].copy()
specific_h=['0']*len(h)

for i,val in enumerate(concept):
    if target[i]=='yes':
        print("instance",i+1,"is positive")
        for x in range(len(specific_h)):
            if specific_h[x]=='0' or specific_h[x]==val[x]:
                specific_h[x]=val[x]
            else:
                specific_h[x]=='?'
        print("the specific hypotesis for training instances",i+1,'is',specific_h)
    if target[i]=='no':
        print('instance',i+1,'is negative hence it is ignored')
        print("the specific hypotesis for training instances",i+1,'is',specific_h)
print("the specific hypotesis for training instances",specific_h)
